001package org.galaxyproject.gxformat2.v19_09.utils;
002
003import java.lang.reflect.Constructor;
004import java.lang.reflect.InvocationTargetException;
005
006public class RecordLoader<T extends Savable> implements Loader<T> {
007  private final Class<? extends T> savableClass;
008
009  public RecordLoader(final Class<? extends T> savableClass) {
010    this.savableClass = savableClass;
011  }
012
013  public T load(
014      final Object doc,
015      final String baseUri,
016      final LoadingOptions loadingOptions,
017      final String docRoot) {
018    Loader.validateOfJavaType(java.util.Map.class, doc);
019    try {
020      final Constructor<? extends T> constructor =
021          this.savableClass.getConstructor(
022              new Class[] {Object.class, String.class, LoadingOptions.class, String.class});
023      final T ret = constructor.newInstance(doc, baseUri, loadingOptions, docRoot);
024      return ret;
025    } catch (InvocationTargetException e) {
026      final Throwable cause = e.getCause();
027      if (cause instanceof RuntimeException) {
028        throw (RuntimeException) cause;
029      }
030      throw new RuntimeException(e.getCause());
031    } catch (ReflectiveOperationException e) {
032      throw new RuntimeException(e);
033    }
034  }
035}

001package org.galaxyproject.gxformat2.v19_09.utils;
002
003public class PrimitiveLoader<T> implements Loader<T> {
004  private Class<T> clazz;
005
006  public PrimitiveLoader(Class<T> clazz) {
007    this.clazz = clazz;
008  }
009
010  public T load(
011      final Object doc,
012      final String baseUri,
013      final LoadingOptions loadingOptions,
014      final String docRoot) {
015    return Loader.validateOfJavaType(this.clazz, doc);
016  }
017}

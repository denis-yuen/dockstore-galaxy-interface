104  public static org.galaxyproject.gxformat2.v19_09.utils.OneOrListOf<
105          org.galaxyproject.gxformat2.v19_09.GalaxyWorkflow>
106      loadDocument(final String doc, final String uri_, final LoadingOptions loadingOptions_) {
107    final String uri = ensureBaseUri(uri_);
108    LoadingOptions loadingOptions = loadingOptions_;
109    if (loadingOptions == null) {
110      loadingOptions = new LoadingOptionsBuilder().setFileUri(uri).build();
111    }
112    final Map<String, Object> result = YamlUtils.mapFromString(doc);
113    loadingOptions.idx.put(uri, result);
114    return loadDocument(result, uri, loadingOptions);
115  }
116
117  static String readPath(final Path path) {
118    try {
119      return new String(Files.readAllBytes(path), "UTF8");
120    } catch (final IOException e) {
121      throw new RuntimeException(e);
122    }
123  }
124
125  static String ensureBaseUri(final String baseUri_) {
126    String baseUri = baseUri_;
127    if (baseUri == null) {
128      baseUri = Uris.fileUri(Paths.get(".").toAbsolutePath().normalize().toString()) + "/";
129    }
130    return baseUri;
131  }

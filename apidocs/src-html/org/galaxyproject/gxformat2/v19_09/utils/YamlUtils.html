001package org.galaxyproject.gxformat2.v19_09.utils;
002
003import java.util.Map;
004import org.yaml.snakeyaml.Yaml;
005
006public class YamlUtils {
007
008  public static Map<String, Object> mapFromString(final String text) {
009    Yaml yaml = new Yaml();
010    final Map<String, Object> result = yaml.load(text);
011    return result;
012  }
013}

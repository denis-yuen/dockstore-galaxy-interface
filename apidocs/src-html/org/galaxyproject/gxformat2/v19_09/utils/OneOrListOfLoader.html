001package org.galaxyproject.gxformat2.v19_09.utils;
002
003import java.util.ArrayList;
004import java.util.List;
005
006public class OneOrListOfLoader<T> implements Loader<OneOrListOf<T>> {
007  private final Loader<T> oneLoader;
008  private final Loader<List<T>> listLoader;
009
010  public OneOrListOfLoader(Loader<T> oneLoader, Loader<List<T>> listLoader) {
011    this.oneLoader = oneLoader;
012    this.listLoader = listLoader;
013  }
014
015  public OneOrListOf<T> load(
016      final Object doc,
017      final String baseUri,
018      final LoadingOptions loadingOptions,
019      final String docRoot) {
020    final List<ValidationException> errors = new ArrayList();
021    try {
022      return OneOrListOf.oneOf(this.oneLoader.load(doc, baseUri, loadingOptions, docRoot));
023    } catch (ValidationException e) {
024      errors.add(e);
025    }
026    try {
027      return OneOrListOf.listOf(this.listLoader.load(doc, baseUri, loadingOptions, docRoot));
028    } catch (ValidationException e) {
029      errors.add(e);
030    }
031    throw new ValidationException("Failed to one or list of of type", errors);
032  }
033}

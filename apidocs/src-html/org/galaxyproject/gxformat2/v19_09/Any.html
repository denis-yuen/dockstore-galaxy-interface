001package org.galaxyproject.gxformat2.v19_09;
002
003import org.galaxyproject.gxformat2.v19_09.utils.ValidationException;
004
005public enum Any {
006  ANY("Any");
007
008  private static String[] symbols = new String[] {"Any"};
009  private String docVal;
010
011  private Any(final String docVal) {
012    this.docVal = docVal;
013  }
014
015  public static Any fromDocumentVal(final String docVal) {
016    for (final Any val : Any.values()) {
017      if (val.docVal.equals(docVal)) {
018        return val;
019      }
020    }
021    throw new ValidationException(String.format("Expected one of %s", Any.symbols, docVal));
022  }
023}

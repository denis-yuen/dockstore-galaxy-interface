001package org.galaxyproject.gxformat2.v19_09;
002
003import org.galaxyproject.gxformat2.v19_09.utils.ValidationException;
004
005public enum WorkflowStepType {
006  TOOL("tool"),
007  SUBWORKFLOW("subworkflow"),
008  PAUSE("pause");
009
010  private static String[] symbols = new String[] {"tool", "subworkflow", "pause"};
011  private String docVal;
012
013  private WorkflowStepType(final String docVal) {
014    this.docVal = docVal;
015  }
016
017  public static WorkflowStepType fromDocumentVal(final String docVal) {
018    for (final WorkflowStepType val : WorkflowStepType.values()) {
019      if (val.docVal.equals(docVal)) {
020        return val;
021      }
022    }
023    throw new ValidationException(
024        String.format("Expected one of %s", WorkflowStepType.symbols, docVal));
025  }
026}

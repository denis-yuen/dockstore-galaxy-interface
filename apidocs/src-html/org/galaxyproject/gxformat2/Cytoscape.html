001package org.galaxyproject.gxformat2;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.HashMap;
006import java.util.List;
007import java.util.Map;
008
009public class Cytoscape {
010  public static final String MAIN_TS_PREFIX = "toolshed.g2.bx.psu.edu/repos/";
011
012  public static List<Map<String, Object>> getElements(final String path) throws Exception {
013    final Map<String, Object> object = (Map<String, Object>) IoUtils.readYamlFromPath(path);
014    return getElements(object);
015  }
016
017  public static List<Map<String, Object>> getElements(final Map<String, Object> object) {
018    final String wfClass = (String) object.get("class");
019    WorkflowAdapter adapter;
020    if (wfClass == null) {
021      adapter = new NativeWorkflowAdapter(object);
022    } else {
023      adapter = new Format2WorkflowAdapter(object);
024    }
025    int orderIndex = 0;
026    final List<Map<String, Object>> elements = new ArrayList<>();
027    for (final WorkflowAdapter.NormalizedStep normalizedStep : adapter.normalizedSteps()) {
028      final Map<String, Object> step = normalizedStep.stepDefinition;
029      String stepId =
030          step.get("label") != null ? (String) step.get("label") : Integer.toString(orderIndex);
031      String stepType = step.get("type") != null ? (String) step.get("type") : "tool";
032      List<String> classes = new ArrayList<>(Arrays.asList("type_" + stepType));
033      if (stepType.equals("tool") || stepType.equals("subworkflow")) {
034        classes.add("runnable");
035      } else {
036        classes.add("input");
037      }
038      String toolId = (String) step.get("tool_id");
039      if (toolId != null && toolId.startsWith(MAIN_TS_PREFIX)) {
040        toolId = toolId.substring(MAIN_TS_PREFIX.length());
041      }
042      String label = normalizedStep.label;
043      if ((label == null || isOrderIndexLabel(label)) && toolId != null) {
044        label = "tool:" + toolId;
045      }
046      final String repoLink;
047      if (step.containsKey("tool_shed_repository")) {
048        final Map<String, String> repo = (Map<String, String>) step.get("tool_shed_repository");
049        repoLink =
050            "https://"
051                + repo.get("tool_shed")
052                + "/view/"
053                + repo.get("owner")
054                + "/"
055                + repo.get("name")
056                + "/"
057                + repo.get("changeset_revision");
058      } else {
059        repoLink = null;
060      }
061      final Map<String, Object> nodeData = new HashMap<String, Object>();
062      nodeData.put("id", stepId);
063      nodeData.put("label", label);
064      nodeData.put("tool_id", step.get("tool_id"));
065      nodeData.put("doc", normalizedStep.doc);
066      nodeData.put("repo_link", repoLink);
067      nodeData.put("step_type", stepType);
068      final Map<String, Object> nodeElement = new HashMap<String, Object>();
069      nodeElement.put("group", "nodes");
070      nodeElement.put("data", nodeData);
071      nodeElement.put("classes", classes);
072      nodeElement.put("position", elementPosition(step));
073      elements.add(nodeElement);
074
075      for (final WorkflowAdapter.Input input : normalizedStep.inputs) {
076        final String edgeId = stepId + "__to__" + input.sourceStepLabel;
077        final Map<String, Object> edgeData = new HashMap<>();
078        edgeData.put("id", edgeId);
079        edgeData.put("source", input.sourceStepLabel);
080        edgeData.put("target", stepId);
081        edgeData.put("input", input.inputName);
082        edgeData.put("output", input.sourceOutputName);
083        final Map<String, Object> edgeElement = new HashMap<>();
084        edgeElement.put("group", "edges");
085        edgeElement.put("data", edgeData);
086        elements.add(edgeElement);
087      }
088      orderIndex++;
089    }
090    return elements;
091  }
092
093  static boolean isOrderIndexLabel(final String label) {
094    try {
095      Integer.parseInt(label);
096      return true;
097    } catch (NumberFormatException e) {
098      return false;
099    }
100  }
101
102  private static Map<String, Long> elementPosition(final Map<String, Object> step) {
103    Map<String, Object> stepPosition = (Map<String, Object>) step.get("position");
104    Map<String, Long> elementPosition = new HashMap();
105    elementPosition.put("x", getIntegerValue(stepPosition, "left"));
106    elementPosition.put("y", getIntegerValue(stepPosition, "top"));
107    return elementPosition;
108  }
109
110  private static Long getIntegerValue(final Map<String, Object> fromMap, final String key) {
111    final Object value = fromMap.get(key);
112    if (value instanceof Float || value instanceof Double) {
113      return (long) Math.floor((double) value);
114    } else if (value instanceof Integer) {
115      return ((Integer) value).longValue();
116    } else {
117      return (long) value;
118    }
119  }

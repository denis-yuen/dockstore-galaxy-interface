018  public static final String MAIN_TS_PREFIX = "toolshed.g2.bx.psu.edu/repos/";
019  public static final String START_ID = "UniqueBeginKey";
020  public static final String END_ID = "UniqueEndKey";

022  public static Map<String, Object> getElements(final String path) throws Exception {
023    final Map<String, Object> object = (Map<String, Object>) IoUtils.readYamlFromPath(path);

027  private static class IdAndLabel {
028    private String id;
029    private String label;
030
031    IdAndLabel(String id, String label) {
032      this.setId(id);
033      this.setLabel(label);
034    }
035
036    public String getId() {
037      return id;
038    }
039
040    public void setId(String id) {
041      this.id = id;
042    }
043
044    public String getLabel() {
045      return label;
046    }
047
048    public void setLabel(String label) {
049      this.label = label;
050    }
051  }

053  public static Map<String, Object> getElements(final Map<String, Object> object) {
054    final String wfClass = (String) object.get("class");
055    WorkflowAdapter adapter;
056    if (wfClass == null) {
057      adapter = new NativeWorkflowAdapter(object);
058    } else {
059      adapter = new Format2WorkflowAdapter(object);
060    }
061    final Map<String, Object> elements = new HashMap<>();
062    final List<Object> nodeElements = new ArrayList<>();
063    final List<Object> edgeElements = new ArrayList<>();
064    elements.put("nodes", nodeElements);
065    elements.put("edges", edgeElements);
066    nodeElements.add(createStartNode());
067    nodeElements.add(createEndNode());
068    List<WorkflowAdapter.NormalizedStep> normalizedSteps = adapter.normalizedSteps();
069    Set<String> endNodeIds =
070        normalizedSteps.stream()
071            .map(normalizedStep -> ((Integer) normalizedStep.stepDefinition.get("id")).toString())
072            .collect(Collectors.toSet());
073    Set<IdAndLabel> allNodesIdsAndLabels =
074        normalizedSteps.stream()
075            .map(
076                normalizedStep -> {
077                  Map<String, Object> stepDefinition = normalizedStep.stepDefinition;
078                  String id = ((Integer) stepDefinition.get("id")).toString();
079                  String label =
080                      stepDefinition.get("label") != null
081                          ? stepDefinition.get("label").toString()
082                          : null;
083                  return new IdAndLabel(id, label);
084                })
085            .collect(Collectors.toSet());
086    for (final WorkflowAdapter.NormalizedStep normalizedStep : adapter.normalizedSteps()) {
087      final Map<String, Object> step = normalizedStep.stepDefinition;
088      String stepId = ((Integer) step.get("id")).toString();
089      String stepType = step.get("type") != null ? (String) step.get("type") : "tool";
090      List<String> classes = new ArrayList<>(Collections.singletonList("type_" + stepType));
091      if (stepType.equals("tool") || stepType.equals("subworkflow")) {
092        classes.add("runnable");
093      } else {
094        classes.add("input");
095      }
096      String toolId = (String) step.get("tool_id");
097      if (toolId != null && toolId.startsWith(MAIN_TS_PREFIX)) {
098        toolId = toolId.substring(MAIN_TS_PREFIX.length());
099      }
100      String label = normalizedStep.label;
101      if ((label == null || isOrderIndexLabel(label)) && toolId != null) {
102        label = "tool:" + toolId;
103      }
104      final String repoLink;
105      if (step.containsKey("tool_shed_repository")) {
106        final Map<String, String> repo = (Map<String, String>) step.get("tool_shed_repository");
107        repoLink =
108            "https://"
109                + repo.get("tool_shed")
110                + "/view/"
111                + repo.get("owner")
112                + "/"
113                + repo.get("name")
114                + "/"
115                + repo.get("changeset_revision");
116      } else {
117        repoLink = null;
118      }
119      final Map<String, Object> nodeData = new HashMap<>();
120      nodeData.put("id", stepId);
121      nodeData.put("label", label);
122      // dockstore displays name, docker, type, tool, and run
123      nodeData.put("name", label);
124      // TODO: detect Docker image properly
125      nodeData.put("docker", "TBD");
126      nodeData.put("run", "TBD");
127
128      nodeData.put("tool_id", step.get("tool_id"));
129      nodeData.put("doc", normalizedStep.doc);
130      nodeData.put("repo_link", repoLink);
131      nodeData.put("step_type", stepType);
132      nodeData.put("type", stepType);
133      final Map<String, Object> nodeElement = new HashMap<>();
134      nodeElement.put("group", "nodes");
135      nodeElement.put("data", nodeData);
136      nodeElement.put("classes", classes);
137      nodeElement.put("position", elementPosition(step));
138      nodeElements.add(nodeElement);
139
140      // Create edge from start node if there's no input connections
141      Object inputConnections = normalizedStep.stepDefinition.get("input_connections");
142      if (normalizedStep.inputs.isEmpty() && inputConnections.toString().equals("{}")) {
143        edgeElements.add(createEdge(START_ID, stepId));
144      }
145      for (final WorkflowAdapter.Input input : normalizedStep.inputs) {
146        String sourceStepLabel = input.sourceStepLabel;
147        final String edgeId = stepId + "__to__" + sourceStepLabel;
148        final Map<String, Object> edgeData = new HashMap<>();
149        edgeData.put("id", edgeId);
150        // Look up what the step ID is based on sourceStepLabel which is either the step ID itself or the step label
151        Optional<IdAndLabel> sourceId =
152            allNodesIdsAndLabels.stream()
153                .filter(
154                    partialStep -> {
155                      boolean isLabel = sourceStepLabel.equals(partialStep.getLabel());
156                      boolean isId = sourceStepLabel.equals(partialStep.getId());
157                      return isLabel || isId;
158                    })
159                .findFirst();
160        if (sourceId.isPresent()) {
161          edgeData.put("source", sourceId.get().id);
162          // Any node that's a source of an edge is not an end node
163          endNodeIds.remove(sourceId.get().id);
164          edgeData.put("target", stepId);
165          edgeData.put("input", input.inputName);
166          edgeData.put("output", input.sourceOutputName);
167          final Map<String, Object> edgeElement = new HashMap<>();
168          edgeElement.put("group", "edges");
169          edgeElement.put("data", edgeData);
170          edgeElements.add(edgeElement);
171        } else {
172          String errorMessage = String.format("Could not find input \"%s\" from the workflow steps.", input.sourceStepLabel);
173          LOG.error(errorMessage);
174        }
175      }
176    }
177    // Create edges for end nodes to direct to the real end node
178    endNodeIds.forEach(endNodeId -> edgeElements.add(createEdge(endNodeId, END_ID)));
179    return elements;
180  }
181
182  static boolean isOrderIndexLabel(final String label) {
183    try {
184      Integer.parseInt(label);
185      return true;
186    } catch (NumberFormatException e) {
187      return false;
188    }
189  }
190
191  private static CytoscapeDAG.Edge createEdge(String source, String target) {
192    String id = source + "__to__" + target;
193    CytoscapeDAG.Edge edge = new CytoscapeDAG.Edge();
194    CytoscapeDAG.Edge.EdgeData edgeData = new CytoscapeDAG.Edge.EdgeData();
195    edgeData.setId(id);
196    edgeData.setSource(source);
197    edgeData.setTarget(target);
198    edge.setData(edgeData);
199    return edge;
200  }
201
202  private static Map<String, Long> elementPosition(final Map<String, Object> step) {
203    Map<String, Object> stepPosition = (Map<String, Object>) step.get("position");
204    Map<String, Long> elementPosition = new HashMap();
205    elementPosition.put("x", getIntegerValue(stepPosition, "left"));
206    elementPosition.put("y", getIntegerValue(stepPosition, "top"));
207    return elementPosition;
208  }
209
210  private static Long getIntegerValue(final Map<String, Object> fromMap, final String key) {
211    final Object value = fromMap.get(key);
212    if (value instanceof Float || value instanceof Double) {
213      return (long) Math.floor((double) value);
214    } else if (value instanceof Integer) {
215      return ((Integer) value).longValue();
216    } else {
217      return (long) value;
218    }
219  }
220
221  private static Map<String, Object> createStartNode() {
222    return createStartOrEndNode(START_ID);
223  }
224
225  private static Map<String, Object> createEndNode() {
226    return createStartOrEndNode(END_ID);
227  }
228
229  private static Map<String, Object> createStartOrEndNode(String id) {
230    CytoscapeDAG.Node.NodeData nodeData = new CytoscapeDAG.Node.NodeData();
231    nodeData.setId(id);
232    nodeData.setName(id);
233    CytoscapeDAG.Node node = new CytoscapeDAG.Node();
234    node.setData(nodeData);
235    return objectMapper.convertValue(node, Map.class);
236  }

001package org.galaxyproject.gxformat2;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.HashMap;
006import java.util.List;
007import java.util.Map;
008
009public class Cytoscape {
010  public static final String MAIN_TS_PREFIX = "toolshed.g2.bx.psu.edu/repos/";
011
012  public static Map<String, Object> getElements(final String path) throws Exception {
013    final Map<String, Object> object = (Map<String, Object>) IoUtils.readYamlFromPath(path);
014    return getElements(object);
015  }
016
017  public static Map<String, Object> getElements(final Map<String, Object> object) {
018    final String wfClass = (String) object.get("class");
019    WorkflowAdapter adapter;
020    if (wfClass == null) {
021      adapter = new NativeWorkflowAdapter(object);
022    } else {
023      adapter = new Format2WorkflowAdapter(object);
024    }
025    int orderIndex = 0;
026    final Map<String, Object> elements = new HashMap<>();
027    final List<Object> nodeElements = new ArrayList<>();
028    final List<Object> edgeElements = new ArrayList<>();
029    elements.put("nodes", nodeElements);
030    elements.put("edges", edgeElements);
031
032    for (final WorkflowAdapter.NormalizedStep normalizedStep : adapter.normalizedSteps()) {
033      final Map<String, Object> step = normalizedStep.stepDefinition;
034      String stepId =
035          step.get("label") != null ? (String) step.get("label") : Integer.toString(orderIndex);
036      String stepType = step.get("type") != null ? (String) step.get("type") : "tool";
037      List<String> classes = new ArrayList<>(Collections.singletonList("type_" + stepType));
038      if (stepType.equals("tool") || stepType.equals("subworkflow")) {
039        classes.add("runnable");
040      } else {
041        classes.add("input");
042      }
043      String toolId = (String) step.get("tool_id");
044      if (toolId != null && toolId.startsWith(MAIN_TS_PREFIX)) {
045        toolId = toolId.substring(MAIN_TS_PREFIX.length());
046      }
047      String label = normalizedStep.label;
048      if ((label == null || isOrderIndexLabel(label)) && toolId != null) {
049        label = "tool:" + toolId;
050      }
051      final String repoLink;
052      if (step.containsKey("tool_shed_repository")) {
053        final Map<String, String> repo = (Map<String, String>) step.get("tool_shed_repository");
054        repoLink =
055            "https://"
056                + repo.get("tool_shed")
057                + "/view/"
058                + repo.get("owner")
059                + "/"
060                + repo.get("name")
061                + "/"
062                + repo.get("changeset_revision");
063      } else {
064        repoLink = null;
065      }
066      final Map<String, Object> nodeData = new HashMap<String, Object>();
067      nodeData.put("id", stepId);
068      nodeData.put("label", label);
069      // dockstore displays name, docker, type, tool, and run
070      nodeData.put("name", label);
071      // TODO: detect Docker image properly
072      nodeData.put("docker", "TBD");
073      nodeData.put("run", "TBD");
074
075      nodeData.put("tool_id", step.get("tool_id"));
076      nodeData.put("doc", normalizedStep.doc);
077      nodeData.put("repo_link", repoLink);
078      nodeData.put("step_type", stepType);
079      final Map<String, Object> nodeElement = new HashMap<String, Object>();
080      nodeElement.put("group", "nodes");
081      nodeElement.put("data", nodeData);
082      nodeElement.put("classes", classes);
083      nodeElement.put("position", elementPosition(step));
084      nodeElements.add(nodeElement);
085
086      for (final WorkflowAdapter.Input input : normalizedStep.inputs) {
087        final String edgeId = stepId + "__to__" + input.sourceStepLabel;
088        final Map<String, Object> edgeData = new HashMap<>();
089        edgeData.put("id", edgeId);
090        edgeData.put("source", input.sourceStepLabel);
091        edgeData.put("target", stepId);
092        edgeData.put("input", input.inputName);
093        edgeData.put("output", input.sourceOutputName);
094        final Map<String, Object> edgeElement = new HashMap<>();
095        edgeElement.put("group", "edges");
096        edgeElement.put("data", edgeData);
097        edgeElements.add(edgeElement);
098      }
099      orderIndex++;
100    }
101
102    return elements;
103  }
104
105  static boolean isOrderIndexLabel(final String label) {
106    try {
107      Integer.parseInt(label);
108      return true;
109    } catch (NumberFormatException e) {
110      return false;
111    }
112  }
113
114  private static Map<String, Long> elementPosition(final Map<String, Object> step) {
115    Map<String, Object> stepPosition = (Map<String, Object>) step.get("position");
116    Map<String, Long> elementPosition = new HashMap();
117    elementPosition.put("x", getIntegerValue(stepPosition, "left"));
118    elementPosition.put("y", getIntegerValue(stepPosition, "top"));
119    return elementPosition;
120  }
121
122  private static Long getIntegerValue(final Map<String, Object> fromMap, final String key) {
123    final Object value = fromMap.get(key);
124    if (value instanceof Float || value instanceof Double) {
125      return (long) Math.floor((double) value);
126    } else if (value instanceof Integer) {
127      return ((Integer) value).longValue();
128    } else {
129      return (long) value;
130    }
131  }

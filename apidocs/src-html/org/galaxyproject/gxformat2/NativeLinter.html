001package org.galaxyproject.gxformat2;
002
003import java.util.ArrayList;
004import java.util.List;
005import java.util.Map;
006
007public class NativeLinter implements GalaxyWorkflowLinter {
008  public static String LINT_FAILED_NO_OUTPUTS = "Workflow contained no outputs";
009  public static String LINT_FAILED_OUTPUT_NO_LABEL = "Workflow contained output without a label";
010
011  public void lint(final LintContext lintContext, final Map<String, Object> workflow) {
012    LintUtils.ensureKey(lintContext, workflow, "format-version", String.class, "0.1");
013    LintUtils.ensureKey(lintContext, workflow, "a_galaxy_workflow", String.class, "true");
014
015    final Map<String, Object> steps =
016        (Map<String, Object>) LintUtils.ensureKey(lintContext, workflow, "steps", Map.class, null);
017    boolean foundOutputs = false;
018    boolean foundOutputsWithoutLabel = false;
019    if (steps != null) {
020      for (Map.Entry<String, Object> stepEntry : steps.entrySet()) {
021        final String orderIndexStr = stepEntry.getKey();
022        try {
023          final int orderIndex = Integer.parseInt(orderIndexStr);
024        } catch (NumberFormatException e) {
025          lintContext.error("expected step_key to be integer not [%s]", orderIndexStr);
026        }
027        if (!(stepEntry.getValue() instanceof Map)) {
028          lintContext.error("expected step value to be Map not [%s]", stepEntry.getValue());
029        }
030        Map<String, Object> step = (Map<String, Object>) stepEntry.getValue();
031        List<String> workflowOutputs =
032            LintUtils.ensureKeyIfPresent(
033                lintContext, step, "workflow_outputs", new ArrayList<String>(), List.class);
034        for (Object workflowOutputObject : workflowOutputs) {
035          foundOutputs = true;
036          if (!(workflowOutputObject instanceof Map)) {
037            lintContext.error("Not a map");
038          }
039          final Map<String, String> workflowOutput = (Map<String, String>) workflowOutputObject;
040          final String label = workflowOutput.get("label");
041          if (label == null || label.length() == 0) {
042            foundOutputsWithoutLabel = true;
043          }
044        }
045        final String stepType =
046            LintUtils.ensureKeyIfPresent(lintContext, step, "type", "tool", String.class);
047        assert stepType != null;
048        if (stepType != null && stepType.equals("subworkflow")) {
049          final Map<String, Object> subworkflow =
050              (Map<String, Object>)
051                  LintUtils.ensureKey(lintContext, step, "subworkflow", Map.class, null);
052          assert subworkflow != null;
053          lint(lintContext, subworkflow);
054        }
055      }
056    }
057    if (!foundOutputs) {
058      lintContext.warn(LINT_FAILED_NO_OUTPUTS);
059    }
060    if (foundOutputsWithoutLabel) {
061      lintContext.warn(LINT_FAILED_OUTPUT_NO_LABEL);
062    }
063  }
064}

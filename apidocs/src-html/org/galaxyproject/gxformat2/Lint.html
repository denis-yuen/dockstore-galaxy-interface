001package org.galaxyproject.gxformat2;
002
003import java.nio.file.Files;
004import java.nio.file.Path;
005import java.nio.file.Paths;
006import java.util.Map;
007import org.yaml.snakeyaml.Yaml;
008
009public class Lint {
010  public static int EXIT_CODE_SUCCESS = 0;
011  public static int EXIT_CODE_LINT_FAILED = 1;
012  public static int EXIT_CODE_FORMAT_ERROR = 2;
013  public static int EXIT_CODE_FILE_PARSE_FAILED = 3;
014
015  public static LintContext lint(final Map<String, Object> workflow) {
016    final String wfClass = (String) workflow.get("class");
017    GalaxyWorkflowLinter linter;
018    if (wfClass != null && wfClass.equals("GalaxyWorkflow")) {
019      linter = new Format2Linter();
020    } else {
021      linter = new NativeLinter();
022    }
023    final LintContext lintContext = new LintContext();
024    linter.lint(lintContext, workflow);
025    return lintContext;
026  }
027
028  public static void main(String[] args) throws Exception {
029    final int exitCode = lint(args);
030    System.exit(exitCode);
031  }
032
033  public static int lint(final String[] args) throws Exception {
034    final Path path = Paths.get(args[0]);
035    final String workflowContents = new String(Files.readAllBytes(path), "UTF8");
036    final Yaml yaml = new Yaml();
037    final Map<String, Object> object = (Map<String, Object>) yaml.load(workflowContents);
038    final LintContext lintContext = lint(object);
039    lintContext.printMessages();
040    int exitCode;
041    if (lintContext.getFoundErrors()) {
042      exitCode = EXIT_CODE_FORMAT_ERROR;
043    } else if (lintContext.getFoundWarns()) {
044      exitCode = EXIT_CODE_LINT_FAILED;
045    } else {
046      exitCode = EXIT_CODE_SUCCESS;
047    }
048    return exitCode;
049  }
050}

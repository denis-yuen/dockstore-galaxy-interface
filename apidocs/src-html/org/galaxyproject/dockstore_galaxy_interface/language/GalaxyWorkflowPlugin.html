020/** @author jmchilton */
021public class GalaxyWorkflowPlugin implements RecommendedLanguageInterface {
022  public static final Logger LOG = LoggerFactory.getLogger(GalaxyWorkflowPlugin.class);
023  public static final String[] TEST_SUFFIXES = {"-tests", "_tests", "-test", "-tests"};
024  public static final String[] TEST_EXTENSIONS = {".yml", ".yaml", ".json"};
025
026  @Override
027  public String launchInstructions(String trsID) {
028    return null;
029  }
030
031  @Override
032  public VersionTypeValidation validateWorkflowSet(
033      String initialPath,
034      String contents,
035      Map<String, Pair<String, GenericFileType>> indexedFiles) {
036    final LintContext lintContext = Lint.lint(loadWorkflow(contents));
037    final boolean valid;
038    if (lintContext.getFoundErrors()) {
039      valid = false;
040    } else {
041      valid = true;
042    }
043    final VersionTypeValidation validation = new VersionTypeValidation(valid, new HashMap<>());
044    return validation;
045  }
046
047  @Override
048  public VersionTypeValidation validateTestParameterSet(
049      Map<String, Pair<String, GenericFileType>> indexedFiles) {
050    return new VersionTypeValidation(true, new HashMap<>());
051  }
052
053  @Override
054  public Pattern initialPathPattern() {
055    // Why doesn't this seem to be called anywhere?
056    return Pattern.compile("/(.*\\.gxwf\\.y[a]?ml|.*\\.ga)");
057  }
058
059  @Override
060  public Map<String, Pair<String, GenericFileType>> indexWorkflowFiles(
061      final String initialPath, final String contents, final FileReader reader) {
062    Map<String, Pair<String, GenericFileType>> results = new HashMap<>();
063    final Optional<String> testParameterFile = findTestParameterFile(initialPath, reader);
064    if (testParameterFile.isPresent()) {
065      results.put(
066          testParameterFile.get(),
067          new ImmutablePair<>(testParameterFile.get(), GenericFileType.TEST_PARAMETER_FILE));
068    }
069    return results;
070  }
071
072  protected Optional<String> findTestParameterFile(
073      final String initialPath, final FileReader reader) {
074    final int extensionPos = initialPath.lastIndexOf(".");
075    final String base = initialPath.substring(0, extensionPos);
076    for (final String suffix : TEST_SUFFIXES) {
077      for (final String extension : TEST_EXTENSIONS) {
078        final String testFile = base + suffix + extension;
079        if (pathExistsFromReader(reader, testFile)) {
080          return Optional.of(testFile);
081        }
082      }
083    }
084    return Optional.empty();
085  }
086
087  protected Boolean pathExistsFromReader(final FileReader reader, final String path) {
088    try {
089      reader.readFile(path);
090      return true;
091    } catch (Exception e) {
092      return false;
093    }
094  }
095
096  static Map<String, Object> loadWorkflow(final String content) {
097    final Yaml yaml = new Yaml();
098    final Map map = yaml.loadAs(content, Map.class);
099    return (Map<String, Object>) map;
100  }
101
102  @Override
103  public RecommendedLanguageInterface.WorkflowMetadata parseWorkflowForMetadata(
104      String initialPath, String content, Map<String, Pair<String, GenericFileType>> indexedFiles) {
105    RecommendedLanguageInterface.WorkflowMetadata metadata =
106        new RecommendedLanguageInterface.WorkflowMetadata();
107    if (content != null && !content.isEmpty()) {
108      try {
109        final Map<String, Object> map = loadWorkflow(content);
110        String name = null;
111        try {
112          name = (String) map.get("name");
113        } catch (ClassCastException e) {
114          LOG.debug("\"name:\" is malformed");
115        }
116
117        // FOLLOWING CODE PULLED FROM cwl handler...
118        String label = null;
119        try {
120          label = (String) map.get("label");
121        } catch (ClassCastException e) {
122          LOG.debug("\"label:\" is malformed");
123        }
124
125        // "doc:" added for CWL 1.0
126        String doc = null;
127        if (map.containsKey("doc")) {
128          Object objectDoc = map.get("doc");
129          if (objectDoc instanceof String) {
130            doc = (String) objectDoc;
131          } else if (objectDoc instanceof Map) {
132            Map docMap = (Map) objectDoc;
133            if (docMap.containsKey("$include")) {
134              String enclosingFile = (String) docMap.get("$include");
135              Optional<Pair<String, GenericFileType>> first =
136                  indexedFiles.values().stream()
137                      .filter(pair -> pair.getLeft().equals(enclosingFile))
138                      .findFirst();
139              if (first.isPresent()) {
140                // No way to fetch this here...
141                LOG.info("$include would have this but reader not passed through, not implemented");
142                doc = null;
143              }
144            }
145          } else if (objectDoc instanceof List) {
146            // arrays for "doc:" added in CWL 1.1
147            List docList = (List) objectDoc;
148            doc = String.join(System.getProperty("line.separator"), docList);
149          }
150        }
151
152        final String finalChoiceForDescription = ObjectUtils.firstNonNull(name, label, doc);
153        if (finalChoiceForDescription != null) {
154          metadata.setDescription(finalChoiceForDescription);
155        } else {
156          LOG.info("Description not found!");
157        }
158
159      } catch (YAMLException | NullPointerException | ClassCastException ex) {
160        String message;
161        if (ex.getCause() != null) {
162          // seems to be possible to get underlying cause in some cases
163          message = ex.getCause().toString();
164        } else {
165          // in other cases, the above will NullPointer
166          message = ex.toString();
167        }
168        LOG.info("Galaxy Workflow file is malformed " + message);
169        // CWL parser gets to put validation information in here,
170        // plugin interface doesn't consume the right information though.
171        // https://github.com/dockstore/dockstore/blob/develop/dockstore-webservice/src/main/java/io/dockstore/webservice/languages/CWLHandler.java#L139
172      }
173    }
174    return metadata;
175  }
